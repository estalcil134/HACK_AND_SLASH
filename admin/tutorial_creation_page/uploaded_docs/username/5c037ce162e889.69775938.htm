#include <c8051_SDCC.h>

#include <stdio.h>

#include <stdlib.h>

#include <i2c.h> //THIS IS THE LIBRARY THAT ALLOWS THE READING FROM AND WRITING TO THE SLAVE REGISTERS

//-----------------------------------------------------------------------------

// 8051 Initialization Functions

//-----------------------------------------------------------------------------

//initializes the PCA so that it is SYSCLK/12, enable interrrupts,

//and then set up the PCA counter/interrupts

void PCA_Init (void);

 

//Initialize the XBR0 Crossbar to be correctly configured for lab 3.2. UART0,

//SPI0 Bus, and the SMBus are all enabled

void XBR0_Init(void);

 

//this initializes the SMBus so that the slave master reads and writes can occur

void SMB_Init(void);

 

//the interrupt system of the PCA has the PCA_Start value, the flag is cleared, and

//then other interrupts are dealt with

void PCA_ISR ( void ) __interrupt 9;

 

//this function correcctly reads the data from the light sensor in register 1 of the slave ultrasonic ranger light sensor

unsigned int ReadRanger(void);

 

 

unsigned int one_count; //this is one 20ms count

unsigned int new_light; //this is the flag that the program must read the light from the ranger (once every 80ms)

unsigned int number_read = 0; //this is the counter that tells the user what number-th read of the light sensor it is

unsigned char Data[1]; //this is an array of length 1 to get the value of the light from the ranger light sensor

 

//-----------------------------------------------------------------------------

// Main Function

//-----------------------------------------------------------------------------

void main(void)

{

              unsigned int light_val; //the printed out light value from the sensor       

 

              Sys_Init(); // initialize board

              putchar(' '); //allows for printf statements

              XBR0_Init(); //initialize the crossbar

              SMB_Init(); //initialize the SMBus

              PCA_Init(); //initialize the PCA

 

              //print beginning message

              printf("Embedded Control Ultrasonic Ranger Light Sensor Section 3\n\r");

 

              while(1)

              {

                             if(new_light) //80ms have passed

                            {

                                           //read the ranger

                                           light_val = ReadRanger();

 

                                           //increment the number of the read (NOT NECESSARY FOR THE LAB BUT I LIKE IT)

                                           number_read++;

 

                                           //start a ping

                                           Data[0] = 0x51; // write 0x51 to reg 0 of the ranger:

                                           i2c_write_data(0xE0,0,Data,1); // write one byte of data to reg 0 at addr

                            

                                           //reset the 80ms flag

                                           new_light = 0;

 

                                           //print the light

                                           printf("The light on the number read: %u, is now: %u\r\n",number_read,light_val);

                             }

              }

}

//-----------------------------------------------------------------------------

// XBR0_Init

//-----------------------------------------------------------------------------

//

// Set up the crossbar

//

void XBR0_Init(void)

{

              XBR0 = 0x27; //configure crossbar with UART0, SPI, SMBus, and CEX3 channels enabled

}

//-----------------------------------------------------------------------------

// SMB_Init

//-----------------------------------------------------------------------------

//

// this initializes the SMBus so that the slave master reads and writes can occur

//

void SMB_Init(void)

{

              SMB0CR = 0x93; //set the SCL to 100kHz

              ENSMB = 1; //bit 6 of SMB0CN, enable the SMBus

}

//-----------------------------------------------------------------------------

// PCA_Init

//-----------------------------------------------------------------------------

//

// Set up Programmable Counter Array

//

void PCA_Init(void)

{

              PCA0MD = 0x81; //set PCA to the sysclck/12

              PCA0CPM3 = 0xC2; //CCM3 in 16 bit compare mode

              PCA0CN = 0x40; //ENABLE PCA COUNTER

              EIE1 |= 0x08; //turn on the PCA interrupt

              EA = 1; //enable the global interrupts

}

//-----------------------------------------------------------------------------

// PCA_ISR

//-----------------------------------------------------------------------------

//

// Interrupt Service Routine for Programmable Counter Array Overflow Interrupt

//

void PCA_ISR (void) __interrupt 9

{

              if(CF) //if the PCA flag is 1

              {

                             CF = 0; //clear the overflow flag

                             PCA0 = 28672; //set the PCA_Start value so that the period is 20ms

                             one_count++; //increment the 20ms counter

                             if(one_count>4)

                             {

                                           new_light = 1; // at least 4 overflows is about 80 ms

                                           one_count = 0; //reset the number of 20ms counts

                             }

              }

              PCA0CN &= 0x40; //deal with other PCA interrupt sources

}

//-----------------------------------------------------------------------------

// ReadRanger

//-----------------------------------------------------------------------------

//

// This function correctly reads the data from the light sensor in register 1 of the slave ultrasonic ranger light sensor

//

unsigned int ReadRanger(void)

{

              unsigned int light; //this is the light value that the is read from the ranger

              unsigned char addr=0xE0; // the address of the ranger is 0xE0

              i2c_read_data(addr,1,Data,1); // read one byte, starting at reg 1

              light = Data[0]; //light becomes the value found from the register

              return light;

}
